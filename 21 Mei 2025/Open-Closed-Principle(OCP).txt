public class DiscountCalculator {
    public double calculateDiscount(String customerType, double amount) {
        if (customerType.equals("Regular")) {
            return amount * 0.1;
        } else if (customerType.equals("Premium")) {
            return amount * 0.2;
        } else {
            return 0;
        }
    }
}

----------------------------------------------------------------------------
public interface iDisc {
    public double getDiscount();
}

public class SilverCust implements iDisc(){
    public double percentage = 0.5;

    public double getDiscount(){
        return percentage;
    }
}

public class GoldCust implements iDisc(){
    public double percentage = 0.7;

    public double getDiscount(){
        return percentage;
    }
}

public class DiscountCalculator {
    public double calculateDiscount(iDisc disc, double amount) {
        return disc.getDiscount() * amount;
    }
}
